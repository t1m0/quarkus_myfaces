document.addEventListener("DOMContentLoaded", function(){ showCorrectActionButton(); }); function edit() { console.log("edit"); showButton('user-form:save-button'); hideButton('edit-button') disableButtons(true); disableFields(false); } function postSave() { console.log("postSave"); hideButton('user-form:save-button'); showButton('edit-button') disableButtons(false); disableFields(true); showCorrectActionButton(); } function showCorrectActionButton() { const state = document.getElementById('user-form:state'); switch (state.value) { case 'New': showButton('user-form:confirm-button') hideButton('user-form:lock-button'); hideButton('user-form:unlock-button'); break; case 'Active': showButton('user-form:lock-button') hideButton('user-form:unlock-button'); hideButton('user-form:confirm-button'); break; case 'Locked': showButton('user-form:unlock-button') hideButton('user-form:lock-button'); hideButton('user-form:confirm-button'); break; } } function hideButton(id) { let button = document.getElementById(id); if(button) { button.classList.add('hidden'); } } function showButton(id) { let button = document.getElementById(id); if(button){ button.classList.remove('hidden'); } } function disableButtons(disabled) { toggleElement('user-form:lock-button', disabled); toggleElement('user-form:unlock-button', disabled); toggleElement('user-form:confirm-button', disabled); } function disableFields(disabled) { let fields = document.querySelectorAll('input[type=\'text\']'); for(const field of fields) { field.disabled = disabled; } } function toggleElement(id, disabled) { let button = document.getElementById(id); if (button) { button.disabled = disabled; } }
